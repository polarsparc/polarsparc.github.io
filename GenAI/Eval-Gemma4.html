from langchain.agents import create_agent
from langchain.chat_models import init_chat_model
from langchain_core.runnables import RunnableConfig

ollama_chat_llm = init_chat_model(ollama_model,
                                  base_url=ollama_base_url,
                                  temperature=llm_temperature,
                                  top_p=llm_top_p,
                                  top_k=llm_top_k)

system_prompt = 'You are a helpful virtual assistant.'

tools_agent = (create_agent(ollama_chat_llm,
                           system_prompt=system_prompt)
               .with_config(RunnableConfig(recursion_limit=3)))

Notice the use of the model provider prefix 'ollama:' along with the model name. This prefix loads and initializes the desired model provider (Ollama in this case).

Test Case - 3 :: Solve a Mathematical Problem


In this test case, we will evaluate the Mathematical problem solving ability of the selected LLM model by prompting a math problem.

Execute the following Python code snippet for the result:


from langchain_core.messages import AIMessage

question = """
If 3x - y = 12, what is the value of (8^x/2^y)?
"""

result = tools_agent.invoke(
  {'messages': [
    {
      'role': 'user',
      'content': question
    }
  ]},
  config={
    'recursion_limit': 3,
    'timeout': llm_timeout
  }
)

agent_messages = [msg for msg in result['messages'] if isinstance(msg, AIMessage)]
if agent_messages:
    print(agent_messages[-1].content)

The following was the typical output:


Output.3

This is a problem that requires using exponent rules and substitution based on the given linear equation.

Here is the step-by-step solution:

### 1. Analyze the Expression and Simplify the Bases

We need to find the value of:
$$\frac{8^x}{2^y}$$

The key to solving this is to express all numbers in the numerator and denominator as powers of the same base, which is **2**.

*   Rewrite $8^x$: Since $8 = 2^3$, we have $8^x = (2^3)^x = 2^{3x}$.
*   The denominator is already in base 2: $2^y$.

Substitute these back into the expression:
$$\frac{2^{3x}}{2^y}$$

### 2. Apply Exponent Rules

When dividing powers with the same base, you subtract the exponents:
$$\frac{2^{3x}}{2^y} = 2^{3x - y}$$

### 3. Use the Given Equation

We are given the equation:
$$3x - y = 12$$

Notice that the exponent of our simplified expression ($3x - y$) is exactly the expression we were given equals 12.

### 4. Solve for the Value

Substitute the value 12 into the simplified expression:
$$2^{3x - y} = 2^{12}$$

Now, calculate $2^{12}$:
$$2^{10} = 1024$$
$$2^{11} = 2048$$
$$2^{12} = 4096$$

---
The value of $\frac{8^x}{2^y}$ is **4096**.

Fantastic - the LLM model was able to successfully solve the math problem !!!

### 1. API Application (`app.py`)

This file sets up the web server, handles the routing, and validates the token for the secure endpoint.

```python
# app.py
from flask import Flask, request, jsonify
import os

app = Flask(__name__)

# --- Configuration ---
# NOTE: In a real application, this token should be loaded from environment variables or a secure vault.
# For this simple example, we are reading it from an environment variable.
# You must set this before running the app: export API_TOKEN="your_secret_token"
API_TOKEN = os.environ.get("API_TOKEN")

if not API_TOKEN:
    print("***********************************************************")
    print("WARNING: API_TOKEN environment variable is not set.")
    print("The /secure-hello endpoint will reject all requests.")
    print("Set it using: export API_TOKEN='<your_generated_token>'")
    print("***********************************************************")


@app.route('/hello', methods=['GET'])
def hello_world():
    """
    Exposes a simple, public 'hello' endpoint.
    """
    return jsonify({
        "message": "Welcome to the public service!",
        "endpoint": "/hello",
        "status": "success"
    })

@app.route('/secure-hello', methods=['GET'])
def secure_hello():
    """
    Exposes a secure endpoint requiring a valid API Token in the Authorization header.
    """
    # 1. Get the token from the request header
    auth_header = request.headers.get('Authorization')
    
    if not auth_header:
        return jsonify({"error": "Authorization header missing. Please provide a token."}), 401
    
    # 2. Expect format: "Bearer <token>"
    try:
        scheme, provided_token = auth_header.split(' ', 1)
        if scheme.lower() != 'bearer':
             return jsonify({"error": "Invalid authorization scheme. Use 'Bearer '."}), 401
    except ValueError:
        return jsonify({"error": "Authorization header must be in the format 'Bearer '."}), 401

    # 3. Validate the token against the configured secret token
    if provided_token == API_TOKEN:
        return jsonify({
            "message": "Success! This is a secure endpoint.",
            "endpoint": "/secure-hello",
            "status": "authenticated"
        })
    else:
        # Token mismatch or missing
        return jsonify({"error": "Invalid API Token provided."}), 403

if __name__ == '__main__':
    print("\n=========================================")
    print("         API Service Started          ")
    print("=========================================")
    print(f"Public Endpoint: http://127.0.0.1:5000/hello")
    print(f"Secure Endpoint: http://127.0.0.1:5000/secure-hello")
    print("-----------------------------------------")
    print("IMPORTANT: Set API_TOKEN environment variable to activate security.")
    app.run(debug=True)
```

### 2. Token Generator Utility (`token_generator.py`)

This standalone script uses Python's `secrets` module for cryptographically secure token generation.

```python
# token_generator.py
import secrets
import string

def generate_api_token(length: int = 32) -> str:
    """
    Generates a cryptographically secure random string suitable for an API token.
    
    Args:
        length: The desired length of the token in characters. Defaults to 32.
        
    Returns:
        The generated token string.
    """
    # Define characters to use in the token (alphanumeric + special chars)
    alphabet = string.ascii_letters + string.digits + string.punctuation
    
    # Use secrets module for secure randomness
    token = ''.join(secrets.choice(alphabet) for _ in range(length))
    return token

if __name__ == "__main__":
    print("--- API Token Generator ---")
    
    # 1. Generate a token
    new_token = generate_api_token()
    
    print(f"\n Successfully generated a new API Token.")
    print(f" Your Secret Token: {new_token}")
    print("-" * 40)
    print("\n Next Steps:")
    print("1. Copy the token displayed above.")
    print("2. In your terminal, set it as an environment variable:")
    print(f"   export API_TOKEN='{new_token}'")
    print("3. Run the API server: python app.py")

```

from langchain.tools import tool

@tool
def execute_shell_command(command: str) -> str:
    """Tool to execute shell commands
    Arg:
        command (str): Shell command to execute
    Returns:
        str: Output of the command execution
    """
    print(f'Executing shell command: {command}')
    try:
        output = subprocess.run(command, shell=True, check=True, text=True, capture_output=True)
        if output.returncode != 0:
            return f'Error executing shell command - {command}'
        return output.stdout
